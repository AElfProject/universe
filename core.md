|\psi\rangle_{\text{低维观察者}}\xrightarrow{\text{经典化}}\text{经典知识（少）}+S_{\text{经典熵（多）}}

|\psi\rangle_{\text{量子意识}}\xrightarrow{\text{经典化}}\text{经典记忆（知识）}+S_{\text{经典熵}}
